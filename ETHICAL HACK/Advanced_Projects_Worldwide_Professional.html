import pefile
import yara
import requests
from sklearn.ensemble import RandomForestClassifier

class MalwareAnalysisEngine:
    def __init__(self):
        self.yara_rules = yara.compile(
            filepath='/path/to/rules'
        )
        self.ml_model = RandomForestClassifier()
    
    def static_analysis(self, file_path):
        # Extract features from PE file
        pe = pefile.PE(file_path)
        features = {
            'imports': self.extract_imports(pe),
            'sections': self.analyze_sections(pe),
            'entropy': self.calculate_entropy(pe),
            'resources': self.analyze_resources(pe),
        }
        return features
    
    def yara_detection(self, file_path):
        # Match against YARA rules
        matches = self.yara_rules.match(file_path)
        return matches
    
    def dynamic_analysis(self, file_path):
        # Submit to Cuckoo Sandbox
        with open(file_path, 'rb') as f:
            response = requests.post(
                'http://cuckoo-server/tasks/create/file',
                files={'file': f}
            )
        return response.json()
    
    def classify_malware(self, features):
        # ML classification: malicious or benign?
        prediction = self.ml_model.predict([features])
        confidence = self.ml_model.predict_proba([features])
        return prediction, confidence

func (ks *K8sSecurityScanner) ScanCluster(ctx context.Context) {
    // Scan all namespaces for security issues
    namespaces, _ := ks.client.CoreV1().Namespaces().List(ctx, metav1.ListOptions{})
    
    for _, ns := range namespaces.Items {
        ks.scanNamespace(ctx, ns.Name)
    }
}

func (ks *K8sSecurityScanner) scanNamespace(
    ctx context.Context, 
    namespace string,
) {
    // Get all pods in namespace
    pods, _ := ks.client.CoreV1().Pods(namespace).List(ctx, metav1.ListOptions{})
    
    for _, pod := range pods.Items {
        // Check for security issues
        ks.checkPodSecurity(&pod)
        ks.checkContainerImages(&pod)
        ks.checkNetworkPolicies(&pod)
    }
}

func (ks *K8sSecurityScanner) checkPodSecurity(
    pod *corev1.Pod,
) {
    for _, container := range pod.Spec.Containers {
        // Check: Running as root?
        if container.SecurityContext.RunAsUser == nil || 
           *container.SecurityContext.RunAsUser == 0 {
            ks.findings = append(ks.findings, SecurityFinding{
                Level: "HIGH",
                Message: "Container running as root",
            })
        }
        
        // Check: Privileged mode?
        if container.SecurityContext.Privileged != nil && 
           *container.SecurityContext.Privileged {
            ks.findings = append(ks.findings, SecurityFinding{
                Level: "CRITICAL",
                Message: "Privileged container detected",
            })
        }
    }
}

import tensorflow as tf
import numpy as np
from cleverhans.attacks import FastGradientMethod

class AdversarialMLSecurity:
    def __init__(self, model):
        self.model = model
        self.fgsm = FastGradientMethod(model)
    
    def generate_adversarial_examples(self, x, y):
        # Generate adversarial examples (fool the model)
        x_adv = self.fgsm.generate(
            x_ph=x,
            y=y,
            eps=0.3,  # Perturbation budget
            ord=np.inf
        )
        return x_adv
    
    def test_model_robustness(self, x_test, y_test):
        # Test model on clean vs adversarial data
        clean_acc = self.model.evaluate(x_test, y_test)[1]
        
        x_adv = self.generate_adversarial_examples(x_test, y_test)
        adv_acc = self.model.evaluate(x_adv, y_test)[1]
        
        robustness = (adv_acc / clean_acc) * 100
        return {
            'clean_accuracy': clean_acc,
            'adversarial_accuracy': adv_acc,
            'robustness_percentage': robustness
        }
    
    def adversarial_training(self, x_train, y_train):
        # Train model with adversarial examples
        for epoch in range(10):
            # Generate adversarial examples
            x_adv = self.generate_adversarial_examples(
                x_train, y_train
            )
            
            # Combine clean + adversarial examples
            x_combined = np.vstack([x_train, x_adv])
            y_combined = np.vstack([y_train, y_train])
            
            # Train on both
            self.model.fit(x_combined, y_combined, epochs=1)
    
    def membership_inference_attack(self, model_output):
        # Attack: Determine if data was in training set
        confidence = tf.nn.softmax(model_output).numpy()
        max_confidence = np.max(confidence)
        
        if max_confidence > 0.9:
            return 'Likely in training set (high confidence)'
        else:
            return 'Likely NOT in training set (low confidence)'

from slither.slither import Slither
from web3 import Web3
import json

class SmartContractAuditor:
    def __init__(self):
        self.slither = Slither
        self.vulnerabilities = []
    
    def audit_contract(self, contract_path):
        # Static analysis using Slither
        slither = self.slither(contract_path)
        
        # Check for common vulnerabilities
        findings = {
            'reentrancy': self.check_reentrancy(slither),
            'integer_overflow': self.check_integer_issues(slither),
            'unchecked_call': self.check_unchecked_calls(slither),
            'access_control': self.check_access_control(slither),
            'delegate_call': self.check_delegate_call(slither),
        }
        
        return findings
    
    def check_reentrancy(self, slither):
        # Most critical vulnerability: Reentrancy
        # Example: The DAO hack ($50M loss)
        reentrancy_vuln = []
        
        for contract in slither.contracts:
            for function in contract.functions:
                # Check for external calls followed by state changes
                calls = function.external_calls_as_expressions
                state_changes = function.state_variables_written
                
                if calls and state_changes:
                    reentrancy_vuln.append({
                        'function': function.name,
                        'severity': 'CRITICAL',
                        'message': 'Potential reentrancy'
                    })
        
        return reentrancy_vuln
    
    def check_access_control(self, slither):
        # Check: Are functions properly protected?
        access_issues = []
        
        for contract in slither.contracts:
            for function in contract.functions:
                # Functions that modify state should be protected
                if function.is_writing_to_state:
                    if not function.has_modifier:
                        access_issues.append({
                            'function': function.name,
                            'issue': 'No access modifier'
                        })
        
        return access_issues
    
    def generate_audit_report(self, findings):
        # Professional audit report
        report = {
            'contract_name': 'Target Contract',
            'audit_date': '2024-06-27',
            'severity_summary': {
                'critical': len([f for f in findings if f.get('severity') == 'CRITICAL']),
                'high': len([f for f in findings if f.get('severity') == 'HIGH']),
            },
            'findings': findings,
            'recommendations': self.generate_recommendations(findings)
        }
        return report
